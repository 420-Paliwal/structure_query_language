1 => unique dept 

SELECT DISTINCT dept
FROM employee
ORDER BY dept;

2 => Highest Salary 

SELECT *
FORM employee
ORDER BY Salary DESC
LIMIT 1;

3 => Unique Salary In HR Department 

SELECT DISTINCT Salary
FROM employee
WHERE dept = 'HR'
ORDER BY Salary;

4 => Employees ko dept wise sort karo,
aur har dept me salary DESC order me

SELECT * FROM employee
ORDER BY dept, Salary DESC

5 => Second highest salary nikaalo (sirf SQL)

SELECT salary FROM employee
ORDER BY Salary DESC
LIMIT 1
OFFSET 1;

6 => Unique (dept, salary) combinations dikhao
 
SELECT DISTINCT dept, Salary 
FROM employee
ORDER BY dept, Salary

7 => IT ya HR dept ke employees jinki salary 40000–60000 ke beech ho
aur naam R se start hota ho

SELECT * 
FROM employee
WHERE (dept = 'IT' OR dept = 'HR')
    AND salary BETWEEN 40000 AND 60000
    AND name LIKE 'R%'

8 => IT ya HR department ke employees

SELECT *
FROM employee
WHERE dept = 'IT' OR dept = 'HR'

9 => Salary 40000 aur 60000 ke beech wale employees

SELECT * 
FROM employee
WHERE salary BETWEEN 40000 AND 60000

10 => Jinka naam 'R' se start hota ho

SELECT *
FROM employee
WHERE name LIKE 'R%'

11 => IT department ke employees jinki salary 50000 se zyada ho
aur naam 'M' se start hota ho

SELECT *
FROM employee
WHERE dept = 'IT'
    AND salary > 50000
    AND name LIKE 'M%'

12 => Employees jo HR ya Finance dept me hain

SELECT *
FROM employee
WHERE dept IN ('HR', 'Finance')

13 => Employees jinki salary NULL hai

SELECT *
FROM employee
WHERE salary IS NULL;

14 => Employees jo IT dept me nahi hain
aur jinki salary 40000 se zyada hai

SELECT *
FROM employee
WHERE dept NOT IN ('IT')
    AND salary > 40000

15 => Total employees kitne hain?

SELECT COUNT(*)
FROM employee

16 => Har department ka total salary

SELECT dept, SUM(salary)
FROM employee
GROUP BY dept;

17 => Sirf wahi departments dikhao
jaha employees ≥ 2 hain

SELECT dept, COUNT(*) AS emp_count
FROM employee 
GROUP BY dept
HAVING COUNT(*) >= 2;

18 => Har department ka highest salary nikaalo

SELECT dept, MAX(salary) AS max_salary
FROM employee
GROUP BY dept;

19 => Kaun sa employee highest salary le raha hai?

SELECT *
FROM employee
WHERE salary = (
    SELECT MAX(salary)
    FROM employee
)
